00070 /**
00071  * Read the value of all the axes registers return by reference
00072  * @param link the HTAC port number
00073  * @param x x axis
00074  * @param y y axis
00075  * @param z z axis
00076  * @return true if no error occured, false if it did
00077  */
00078 bool HTACreadAllAxes(tSensors link, int &x, int &y, int &z) {
00079   memset(HTAC_I2CRequest, 0, sizeof(tByteArray));
00080 
00081   HTAC_I2CRequest.arr[0] = 2;                       // Message size
00082   HTAC_I2CRequest.arr[1] = HTAC_I2C_ADDR;           // I2C Address
00083   HTAC_I2CRequest.arr[2] = HTAC_OFFSET + HTAC_X_UP; // X axis upper 8 bits register
00084 
00085   if (!writeI2C(link, HTAC_I2CRequest, 6))
00086     return false;
00087 
00088   if (!readI2C(link, HTAC_I2CReply, 6))
00089     return false;
00090 
00091   // Convert 2 bytes into a signed 10 bit value.  If the 8 high bits are more than 127, make
00092   // it a signed value before combing it with the lower 2 bits.
00093   // Gotta love conditional assignments!
00094   x = (HTAC_I2CReply.arr[0] > 127) ? (HTAC_I2CReply.arr[0] - 256) * 4 + HTAC_I2CReply.arr[3]
00095                                    : HTAC_I2CReply.arr[0] * 4 + HTAC_I2CReply.arr[3];
00096 
00097   y = (HTAC_I2CReply.arr[1] > 127) ? (HTAC_I2CReply.arr[1] - 256) * 4 + HTAC_I2CReply.arr[4]
00098                                    : HTAC_I2CReply.arr[1] * 4 + HTAC_I2CReply.arr[4];
00099 
00100   z = (HTAC_I2CReply.arr[2] > 127) ? (HTAC_I2CReply.arr[2] - 256) * 4 + HTAC_I2CReply.arr[5]
00101                                    : HTAC_I2CReply.arr[2] * 4 + HTAC_I2CReply.arr[5];
00102 
00103   return true;
00104 }
