00072 /**

00074  * @param link the HTIRS port number
00075  * @return value of 0-9, the direction index of the detected IR signal or -1 if an error occurred.
00076  */
00077 int HTIRSreadDir(tSensors link) {
00078   memset(HTIRS_I2CRequest, 0, sizeof(tByteArray));
00079 
00080   HTIRS_I2CRequest.arr[0] = 2;                        // Message size
00081   HTIRS_I2CRequest.arr[1] = HTIRS_I2C_ADDR;           // I2C Address
00082   HTIRS_I2CRequest.arr[2] = HTIRS_OFFSET + HTIRS_DIR; // Start direction register
00083 
00084   if (!writeI2C(link, HTIRS_I2CRequest, 1))
00085     return -1;
00086 
00087   if (!readI2C(link, HTIRS_I2CReply, 1))

00089
